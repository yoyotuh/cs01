### ‚úÖ Python Code Example ‚Äî Complete Mini Project

```python
# Student Grades Analyzer

students = []

def add_student(name, grade):
    students.append({"name": name, "grade": grade})

def search_student(name):
    for student in students:
        if student["name"].lower() == name.lower():
            return student
    return None

def count_above(threshold):
    count = 0
    for student in students:
        if student["grade"] > threshold:
            count += 1
    return count

def statistics():
    if not students:
        return None
    grades = [s["grade"] for s in students]
    return {
        "max": max(grades),
        "min": min(grades),
        "average": sum(grades)/len(grades)
    }

# Sample data
add_student("Alice", 90)
add_student("Bob", 75)
add_student("Charlie", 85)

# User interaction
try:
    name = input("Enter student name to search: ")
    result = search_student(name)
    if result:
        print(f"{result['name']} ‚Üí {result['grade']}")
    else:
        print("Student not found.")
except Exception as e:
    print("Error:", e)

print("Students with grade > 80:", count_above(80))
stats = statistics()
print("Statistics:", stats)
```
